## <a name="文件下载">文件下载</a>
```js
// 模拟鼠标点击事件
function autoClick(el) {
  if (document.createEvent){
    // 自定义点击事件
    let DownloadEvt = document.createEvent('MouseEvents');
    DownloadEvt.initEvent('click', true, false);
    el.dispatchEvent(DownloadEvt);
  }else if (document.createEventObject){
    el.fireEvent('onclick');
  }else if (typeof el.onclick == 'function' ){
    el.onclick();
  }
}

// 下载
function downloadFile(downUrl,filename){
  let a = document.createElement('a');
  if(a) {
    a.style = 'display: none';
    a.download = filename;
    a.href = downUrl;
    console.log(a);
    document.body.appendChild(a);
    autoClick(a)
    document.body.removeChild(a);
  }
}
```

例：json下载
```js
const json = {
  a: 3,
  b: 4,
  c: 5
}
const str = JSON.stringify(json, null, 2)

// 方案一：Text -> DataURL
// json 视为字符串，可以利用 DataURL 进行下载
const dataUrl = `data:,${str}`
downloadFile(dataUrl, 'demo.json')

// 方案二：Text -> Blob -> ObjectURL
// json 转换为 Blob 再转化为 ObjectURL 来构造 URL
const url = URL.createObjectURL(new Blob(str.split('')))
downloadFile(url, 'demo1.json')
```

## xlsx下载
```html
<template>
  <el-button @click="download" type="primary">下载示例表</el-button>
</template>
<script>
import xslx from "xlsx";

export default {
  name: '',
  components: {},
  data () {
    return {
      tableData: [
        {
          Name: 'name1',
          IDCard: 1,
          Phone: 1
        },
        {
          Name: 'name2',
          IDCard: 2,
          Phone: 2
        },
        {
          Name: 'name3',
          IDCard: 3,
          Phone: 3
        },
      ],
    }
  },
  created () {},
  methods: {
    download() {
      let arr = this.tableData.map(item => {
        return {
          '姓名': item.Name,
          '手机号': item.Phone,
          '身份证': item.IDCard,
        };
      });
      let sheet = xslx.utils.json_to_sheet(arr),
        book = xslx.utils.book_new();
      xslx.utils.book_append_sheet(book, sheet, "Sheet1");
      xslx.writeFile(book, `用户数据实例表.xlsx`);
    }
  }
}
</script>
```
