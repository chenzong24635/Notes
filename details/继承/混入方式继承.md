# <a name="混入方式继承">混入方式继承多个对象</a>
通过原型拷贝其他类的方法属性

```js
function Super1(name = 'super1', colors1 = ["red"]) {
  this.colors1 = colors1
  this.name = name
}
function Super2(name = 'super2', colors2 = ["black"]) {
  this.colors2 = colors2
  this.name = name
}

function Sub(name = 'sub' ,age) {
  Super1.call(this, name);
  Super2.call(this, name);
  this.age = age
}

Sub.prototype = Object.create(Super1.prototype);// 继承一个类
Object.assign(Sub.prototype, Super2.prototype);// 混合其它类

// 重新指定constructor
Sub.prototype.constructor = Sub;

Sub.prototype.sayName = function() {
  console.log(this.name);
};

let sub1 = new Sub('sub1', 11);
let sub2 = new Sub('sub2', 22);

sub1.colors1.push('#fff')
console.log(sub1.colors1); // ["red", "#fff"]
console.log(sub2.colors1); // ["red"]

sub1.sayName(); // 'sub1'
sub2.sayName(); // 'sub2'

console.log(sub1.age); // 11
console.log(sub2.age); // 22

console.log(sub1 instanceof Super1) //true
console.log(sub1 instanceof Super2) //false
console.log(sub1 instanceof Sub) //true
```
